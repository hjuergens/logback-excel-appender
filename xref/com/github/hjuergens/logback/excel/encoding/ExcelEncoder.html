15   */
16  public class ExcelEncoder<E> extends LayoutWrappingEncoder<E> {
17  
18      private Workbook workbook;
19      private String fileName;
20      private String password;
21      private String sheetName;
22  
23      public ExcelEncoder() {
24      }
25  
26      @Override
27      public void init(OutputStream os) throws IOException {
28          super.init(os);
29          this.workbook = (Workbook) getContext().getObject("workbook");
30          Workbook workbook = null;
31          try {
32              File file = new File(fileName);
33              workbook = WorkbookFactory.create(file, password);
34              workbook.createSheet(sheetName);
35              WorkbookUtil.createSafeSheetName(fileName);
36              this.context.putObject("workbench", workbook);
37          } catch (InvalidFormatException e) {
38              workbook = new SXSSFWorkbook(100); // keep 100 rows in memory, exceeding rows will be flushed to disk
39              this.context.putObject("workbench", workbook);
40              throw new IOException(e.getMessage());
41          }
42      }
43  
44      @Override
45      public void doEncode(E event) throws IOException {
46          String val = layout.doLayout(event);
47  
48              /*
49              Sheet sh = wb.createSheet();
50              for(int rownum = 0; rownum < 1000; rownum++){
51                  Row row = sh.createRow(rownum);
52                  for(int cellnum = 0; cellnum < 10; cellnum++){
53                      Cell cell = row.createCell(cellnum);
54                      String address = new CellReference(cell).formatAsString();
55                      cell.setCellValue(address);
56                  }
57              }
58              */
59  
60          outputStream.write(val.getBytes());
61          // necessary if ResilientFileOutputStream is buffered
62          outputStream.flush();
63      }
64  
65      @Override
66      public void close() throws IOException {
67          workbook.write(outputStream);
68          this.outputStream.close();
69      }
70  
71      public void setFileName(String fileName) {
72          this.fileName = fileName;
73      }
74  
75      public void setSheetName(String sheetName) {
76          this.sheetName = sheetName;
77      }
78  
79      public void setPassword(String password) {
80          this.password = password;
81      }
82  }
